32 /** \brief Get the Jacobian elements used by a particular reaction
33  *
34  * \param model_data A pointer to the model data
35  * \param jac Jacobian
36  */
37 void rxn_get_used_jac_elem(ModelData *model_data, Jacobian *jac) {
38  // Get the number of reactions
39  int n_rxn = model_data->n_rxn;
40 
41  // Loop through the reactions to determine the Jacobian elements used
42  // advancing the rxn_data pointer each time
43  for (int i_rxn = 0; i_rxn < n_rxn; i_rxn++) {
44  // Get pointers to the reaction data
45  int *rxn_int_data =
46  &(model_data->rxn_int_data[model_data->rxn_int_indices[i_rxn]]);
47  double *rxn_float_data =
48  &(model_data->rxn_float_data[model_data->rxn_float_indices[i_rxn]]);
49 
50  // Get the reaction type
51  int rxn_type = *(rxn_int_data++);
52 
53  // Call the appropriate function
54  switch (rxn_type) {
55  case RXN_AQUEOUS_EQUILIBRIUM:
56  rxn_aqueous_equilibrium_get_used_jac_elem(rxn_int_data, rxn_float_data,
57  jac);
58  break;
59  case RXN_ARRHENIUS:
60  rxn_arrhenius_get_used_jac_elem(rxn_int_data, rxn_float_data, jac);
61  break;
62  case RXN_CMAQ_H2O2:
63  rxn_CMAQ_H2O2_get_used_jac_elem(rxn_int_data, rxn_float_data, jac);
64  break;
65  case RXN_CMAQ_OH_HNO3:
66  rxn_CMAQ_OH_HNO3_get_used_jac_elem(rxn_int_data, rxn_float_data, jac);
67  break;
68  case RXN_CONDENSED_PHASE_ARRHENIUS:
69  rxn_condensed_phase_arrhenius_get_used_jac_elem(rxn_int_data,
70  rxn_float_data, jac);
71  break;
72  case RXN_EMISSION:
73  rxn_emission_get_used_jac_elem(rxn_int_data, rxn_float_data, jac);
74  break;
75  case RXN_FIRST_ORDER_LOSS:
76  rxn_first_order_loss_get_used_jac_elem(rxn_int_data, rxn_float_data,
77  jac);
78  break;
79  case RXN_HL_PHASE_TRANSFER:
80  rxn_HL_phase_transfer_get_used_jac_elem(model_data, rxn_int_data,
81  rxn_float_data, jac);
82  break;
83  case RXN_PHOTOLYSIS:
84  rxn_photolysis_get_used_jac_elem(rxn_int_data, rxn_float_data, jac);
85  break;

527  *
528  * \param rxn_type Reaction type
529  * \param n_int_param Number of integer parameters
530  * \param n_float_param Number of floating-point parameters
531  * \param n_env_param Number of environment-dependent parameters
532  * \param int_param Pointer to integer parameter array
533  * \param float_param Pointer to floating-point parameter array
534  * \param solver_data Pointer to solver data
535  */
536 // TODO: question: move n_added_rxns out of struct to function parameter since
537 // is only used in this function
538 void rxn_add_condensed_data(int rxn_type, int n_int_param, int n_float_param,
539  int n_env_param, int *int_param,
540  double *float_param, void *solver_data) {
541  ModelData *model_data =
542  (ModelData *)&(((SolverData *)solver_data)->model_data);
543 
544  // Get pointers to the reaction data
545  int *rxn_int_data =
546  &(model_data->rxn_int_data
547  [model_data->rxn_int_indices[model_data->n_added_rxns]]);
548  double *rxn_float_data =
549  &(model_data->rxn_float_data
550  [model_data->rxn_float_indices[model_data->n_added_rxns]]);
551 
552  // Save next indices by adding lengths
553  model_data->rxn_int_indices[model_data->n_added_rxns + 1] =
554  (n_int_param + 1) +
555  model_data->rxn_int_indices[model_data->n_added_rxns]; //+1 is type
556  model_data->rxn_float_indices[model_data->n_added_rxns + 1] =
557  n_float_param + model_data->rxn_float_indices[model_data->n_added_rxns];
558  model_data->rxn_env_idx[model_data->n_added_rxns + 1] =
559  model_data->rxn_env_idx[model_data->n_added_rxns] + n_env_param;
560  ++(model_data->n_added_rxns);
561 
562  // Add the reaction type
563  *(rxn_int_data++) = rxn_type;
564 
565  // Add integer parameters
566  for (; n_int_param > 0; --n_int_param) *(rxn_int_data++) = *(int_param++);
567 
568  // Add floating-point parameters
569  for (; n_float_param > 0; --n_float_param)
570  *(rxn_float_data++) = (double)*(float_param++);
571 
572  model_data->n_rxn_env_data += n_env_param;
573 }

575 /** \brief Update reaction data
576  *
577  * Update data for one or more reactions. Reactions of a certain type are
578  * passed a void pointer to updated data that must be in the format specified
579  * by the reaction type. This data could be used to find specific reactions of
580  * the specified type and, for example, update rate constants.
581  *
582  * \param cell_id Id of the grid cell to update
583  * \param rxn_id Id of the reaction (or 0 if unknown)
584  * \param update_rxn_type Type of the reaction
585  * \param update_data Pointer to updated data to pass to the reaction
586  * \param solver_data Pointer to solver data
587  */
588 void rxn_update_data(int cell_id, int *rxn_id, int update_rxn_type,
589  void *update_data, void *solver_data) {
590  ModelData *model_data =
591  (ModelData *)&(((SolverData *)solver_data)->model_data);
592 
593  // Point to the environment-dependent data for the grid cell
594  model_data->grid_cell_rxn_env_data =
595  &(model_data->rxn_env_data[cell_id * model_data->n_rxn_env_data]);
596 
597  // Get the number of reactions
598  int n_rxn = model_data->n_rxn;
599 
600  // Loop through the reactions advancing the rxn_data pointer each time
601  for (; (*rxn_id) < n_rxn; (*rxn_id)++) {
602  // Get pointers to the reaction data
603  int *rxn_int_data =
604  &(model_data->rxn_int_data[model_data->rxn_int_indices[*rxn_id]]);
605  double *rxn_float_data =
606  &(model_data->rxn_float_data[model_data->rxn_float_indices[*rxn_id]]);
607 
608  double *rxn_env_data =
609  &(model_data->grid_cell_rxn_env_data[model_data->rxn_env_idx[*rxn_id]]);
610 
611  // Get the reaction type
612  int rxn_type = *(rxn_int_data++);
613 
614  bool found = false;
615 
616  // Try the update data function for reactions of the correct type
617  if (rxn_type == update_rxn_type) {
618  switch (rxn_type) {
619  case RXN_EMISSION:
620  found = rxn_emission_update_data((void *)update_data, rxn_int_data,
621  rxn_float_data, rxn_env_data);
622  break;
623  case RXN_FIRST_ORDER_LOSS:
624  found = rxn_first_order_loss_update_data(
625  (void *)update_data, rxn_int_data, rxn_float_data, rxn_env_data);
626  break;
627  case RXN_PHOTOLYSIS:
628  found = rxn_photolysis_update_data((void *)update_data, rxn_int_data,
629  rxn_float_data, rxn_env_data);
630  break;
631  case RXN_WET_DEPOSITION:
632  found = rxn_wet_deposition_update_data(
633  (void *)update_data, rxn_int_data, rxn_float_data, rxn_env_data);
634  break;
635  }
636  if (found) return;
637  }
638  }
639 }

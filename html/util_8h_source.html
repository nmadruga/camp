17 /** Calculate the mean free path of a gas-phase species [m]
18  * \f[
19  * \lambda = 3.0 D_g / v
20  * \f]
21  * where \f$D_g\f$ is the gas-phase diffusion coefficient
22  * [\f$\mbox{m}^2\,\mbox{s}^{-1}\f$] and
23  * \f$v\f$ is the mean speed of the gas-phase molecules
24  * [ \f$\mbox{m}\,\mbox{s}^{-1}\f$ ] :
25  * \f[
26  * v = \sqrt{\frac{8RT}{\pi MW}}
27  * \f]
28  * where R is the ideal gas constant [\f$\mbox{J}\, \mbox{K}^{-1}\,
29  * \mbox{mol}^{-1}\f$], T is temperature [K], and MW is the molecular weight of
30  * the gas-phase species
31  * [\f$\mbox{kg}\, \mbox{mol}^{-1}\f$]
32  *
33  * @param diffusion_coeff__m2_s Diffusion coefficient of the gas species
34  * [\f$\mbox{m}^2\, \mbox{s}^{-1}\f$]
35  * @param temperature__K Temperature [K]
36  * @param mw__kg_mol Molecular weight of the gas-phase species [\f$\mbox{kg}\,
37  * \mbox{mol}^{-1}\f$]
38  */
39 static inline double mean_free_path__m(double diffusion_coeff__m2_s,
40  double temperature__K,
41  double mw__kg_mol) {
42  return 3.0 * diffusion_coeff__m2_s /
43  (sqrt(8.0 * UNIV_GAS_CONST_ * temperature__K / (M_PI * mw__kg_mol)));
44 }

46 /** Calculate the transition regime correction factor [unitless] \cite Fuchs1971
47  * : \f[ f(K_n,\alpha) = \frac{0.75 \alpha ( 1 + K_n )}{K_n(1+K_n) + 0.283\alpha
48  * K_n + 0.75 \alpha} \f] where the Knudsen Number \f$K_n = \lambda / r\f$
49  * (where \f$\lambda\f$ is the mean free path [m] of the gas-phase species and
50  * \f$r\f$ is the effective radius of the particles [m]), and \f$ \alpha \f$ is
51  * the mass accomodation coefficient, which is typically assumed to equal 0.1
52  * \cite Zaveri2008.
53  *
54  * @param mean_free_path__m mean free path of the gas-phase species [m]
55  * @param radius__m Particle effective radius [m]
56  * @param alpha Mass accomodation coefficient [unitless]
57  */
58 static inline double transition_regime_correction_factor(
59  double mean_free_path__m, double radius__m, double alpha) {
60  double K_n = mean_free_path__m / radius__m;
61  return (0.75 * alpha * (1.0 + K_n)) /
62  (K_n * K_n + (1.0 + 0.283 * alpha) * K_n + 0.75 * alpha);
63 }

65 /** Calculate the derivative of the transition regime correction factor by
66  * particle radius.
67  * \f[
68  * \frac{d f_{fs}}{d r} = \frac{0.75 \alpha \lambda (K_n^2 + 2K_n + 1 +
69  * (0.283-0.75)\alpha)}{r^2(K_n^2 + (1+0.283\alpha)K_n + 0.75\alpha)^2} \f]
70  * \todo double check the correction factor derivative equation
71  * where the Knudsen Number \f$K_n = \lambda / r\f$ (where \f$\lambda\f$ is the
72  * mean free path [m] of the gas-phase species and \f$r\f$ is the effective
73  * radius of the particles [m]), and \f$ \alpha \f$ is the mass accomodation
74  * coefficient, which is typically assumed to equal 0.1 \cite Zaveri2008.
75  *
76  * @param mean_free_path__m mean free path of the gas-phase species [m]
77  * @param radius__m Particle effective radius [m]
78  * @param alpha Mass accomodation coefficient [unitless]
79  */
80 static inline double d_transition_regime_correction_factor_d_radius(
81  double mean_free_path__m, double radius__m, double alpha) {
82  double K_n = mean_free_path__m / radius__m;
83  return (0.75 * alpha * mean_free_path__m *
84  (K_n * K_n + 2.0 * K_n + 1.0 + (0.283 - 0.75) * alpha)) /
85  pow(radius__m *
86  (K_n * K_n + (1.0 + 0.283 * alpha) * K_n + 0.75 * alpha),
87  2);
88 }

90 /** Calculate the gas-aerosol reaction rate constant for the transition regime
91  * [\f$\mbox{m}^3\, \mbox{particle}^{-1}\, \mbox{s}^{-1}\f$]
92  *
93  * The rate constant \f$k_c\f$ is calculated according to \cite Zaveri2008 as:
94  * \f[
95  * k_c = 4 \pi r D_g f_{fs}( K_n, \alpha )
96  * \f]
97  * where \f$r\f$ is the radius of the particle(s) [m], \f$D_g\f$ is the
98  * diffusion coefficient of the gas-phase species
99  * [\f$\mbox{m}^2\,\mbox{s}^{-1}\f$], \f$f_{fs}( K_n, \alpha )\f$ is the Fuchs
100  * Sutugin transition regime correction factor [unitless], \f$K_n\f$ is the
101  * Knudsen Number [unitess], and \f$\alpha\f$ is the mass accomodation
102  * coefficient.
103  *
104  * Rates can be calculated as:
105  * \f[
106  * r_c = [G] N_a k_c
107  * \f]
108  * where \f$[G]\f$ is the gas-phase species concentration [ppm], \f$N_a\f$ is
109  * the number concentration of particles [\f$\mbox{particle}\,\mbox{m}^{-3}\f$]
110  * and the rate \f$r_c\f$ is in [\f$\mbox{ppm}\,\mbox{s}^{-1}\f$].
111  *
112  * @param diffusion_coeff__m2_s Diffusion coefficent of the gas species
113  * [\f$\mbox{m}^2\, \mbox{s}^{-1}\f$]
114  * @param mean_free_path__m Mean free path of gas molecules [m]
115  * @param radius__m Particle radius [m]
116  * @param alpha Mass accomodation coefficient [unitless]
117  */
118 static inline double gas_aerosol_rxn_rate_constant(double diffusion_coeff__m2_s,
119  double mean_free_path__m,
120  double radius__m,
121  double alpha) {
122  return 4.0 * M_PI * radius__m * diffusion_coeff__m2_s *
123  transition_regime_correction_factor(mean_free_path__m, radius__m,
124  alpha);
125 }

127 /** Calculate the derivative of a gas-aerosol reaction rate by particle radius
128  * \f[
129  * \frac{d k_c}{d r} = 4 \pi D_g ( f_{fs} + r \frac{d_{fs}}{d r} )
130  * \f]
131  * where \f$r\f$ is the radius of the particle(s) [m], \f$D_g\f$ is the
132  * diffusion coefficient of the gas-phase species [\f$\mbox{m}^2\,
133  * \mbox{s}^{-1}\f$] and \f$f_{fs}( K_n, \alpha )\f$ is the Fuchs Sutugin
134  * transition regime correction factor [unitless] (\f$K_n\f$ is the Knudsen
135  * Number [unitess] and \f$\alpha\f$ is the mass accomodation coefficient.
136  *
137  * @param diffusion_coeff__m2_s Diffusion coefficent of the gas species
138  * [\f$\mbox{m}^2\, \mbox{s}^{-1}\f$]
139  * @param mean_free_path__m Mean free path of gas molecules [m]
140  * @param radius__m Particle radius [m]
141  * @param alpha Mass accomodation coefficient [unitless]
142  */
143 static inline double d_gas_aerosol_rxn_rate_constant_d_radius(
144  double diffusion_coeff__m2_s, double mean_free_path__m, double radius__m,
145  double alpha) {
146  return 4.0 * M_PI * diffusion_coeff__m2_s *
147  (transition_regime_correction_factor(mean_free_path__m, radius__m,
148  alpha) +
149  radius__m * d_transition_regime_correction_factor_d_radius(
150  mean_free_path__m, radius__m, alpha));
151 }

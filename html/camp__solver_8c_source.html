364 /** \brief Solver initialization
365  *
366  * Allocate and initialize solver objects
367  *
368  * \param solver_data Pointer to a SolverData object
369  * \param abs_tol Pointer to array of absolute tolerances
370  * \param rel_tol Relative integration tolerance
371  * \param max_steps Maximum number of internal integration steps
372  * \param max_conv_fails Maximum number of convergence failures
373  */
374 void solver_initialize(void *solver_data, double *abs_tol, double rel_tol,
375  int max_steps, int max_conv_fails) {
376 #ifdef CAMP_USE_SUNDIALS
377  SolverData *sd; // SolverData object
378  int flag; // return code from SUNDIALS functions
379  int n_dep_var; // number of dependent variables per grid cell
380  int i_dep_var; // index of dependent variables in loops
381  int n_state_var; // number of variables on the state array per
382  // grid cell
383  int n_cells; // number of cells to solve simultaneously
384  int *var_type; // state variable types
385 
386  // Seed the random number generator
387  srand((unsigned int)100);
388 
389  // Get a pointer to the SolverData
390  sd = (SolverData *)solver_data;
391 
392  // Create a new solver object
393  sd->cvode_mem = CVodeCreate(CV_BDF, CV_NEWTON);
394  check_flag_fail((void *)sd->cvode_mem, "CVodeCreate", 0);
395 
396  // Get the number of total and dependent variables on the state array,
397  // and the type of each state variable. All values are per-grid-cell.
398  n_state_var = sd->model_data.n_per_cell_state_var;
399  n_dep_var = sd->model_data.n_per_cell_dep_var;
400  var_type = sd->model_data.var_type;
401  n_cells = sd->model_data.n_cells;
402 
403  // Set the solver data
404  flag = CVodeSetUserData(sd->cvode_mem, sd);
405  check_flag_fail(&flag, "CVodeSetUserData", 1);
406 
407  /* Call CVodeInit to initialize the integrator memory and specify the
408  * right-hand side function in y'=f(t,y), the initial time t0, and
409  * the initial dependent variable vector y. */
410  flag = CVodeInit(sd->cvode_mem, f, (realtype)0.0, sd->y);
411  check_flag_fail(&flag, "CVodeInit", 1);
412 
413  // Set the relative and absolute tolerances
414  sd->abs_tol_nv = N_VNew_Serial(n_dep_var * n_cells);
415  i_dep_var = 0;
416  for (int i_cell = 0; i_cell < n_cells; ++i_cell)
417  for (int i_spec = 0; i_spec < n_state_var; ++i_spec)
418  if (var_type[i_spec] == CHEM_SPEC_VARIABLE)
419  NV_Ith_S(sd->abs_tol_nv, i_dep_var++) = (realtype)abs_tol[i_spec];
420  flag = CVodeSVtolerances(sd->cvode_mem, (realtype)rel_tol, sd->abs_tol_nv);
421  check_flag_fail(&flag, "CVodeSVtolerances", 1);
422 
423  // Add a pointer in the model data to the absolute tolerances for use during
424  // solving. TODO find a better way to do this
425  sd->model_data.abs_tol = abs_tol;
426 
427  // Set the maximum number of iterations
428  flag = CVodeSetMaxNumSteps(sd->cvode_mem, max_steps);
429  check_flag_fail(&flag, "CVodeSetMaxNumSteps", 1);
430 
431  // Set the maximum number of convergence failures
432  flag = CVodeSetMaxConvFails(sd->cvode_mem, max_conv_fails);
433  check_flag_fail(&flag, "CVodeSetMaxConvFails", 1);
434 
435  // Set the maximum number of error test failures (TODO make separate input?)
436  flag = CVodeSetMaxErrTestFails(sd->cvode_mem, max_conv_fails);
437  check_flag_fail(&flag, "CVodeSetMaxErrTestFails", 1);
438 
439  // Set the maximum number of warnings about a too-small time step
440  flag = CVodeSetMaxHnilWarns(sd->cvode_mem, MAX_TIMESTEP_WARNINGS);
441  check_flag_fail(&flag, "CVodeSetMaxHnilWarns", 1);
442 
443  // Get the structure of the Jacobian matrix
444  sd->J = get_jac_init(sd);
445  sd->model_data.J_init = SUNMatClone(sd->J);
446  SUNMatCopy(sd->J, sd->model_data.J_init);
447 
448  // Create a Jacobian matrix for correcting negative predicted concentrations
449  // during solving
450  sd->J_guess = SUNMatClone(sd->J);
451  SUNMatCopy(sd->J, sd->J_guess);
452 
453  // Create a KLU SUNLinearSolver
454  sd->ls = SUNKLU(sd->y, sd->J);
455  check_flag_fail((void *)sd->ls, "SUNKLU", 0);
456 
457  // Attach the linear solver and Jacobian to the CVodeMem object
458  flag = CVDlsSetLinearSolver(sd->cvode_mem, sd->ls, sd->J);
459  check_flag_fail(&flag, "CVDlsSetLinearSolver", 1);
460 
461  // Set the Jacobian function to Jac
462  flag = CVDlsSetJacFn(sd->cvode_mem, Jac);
463  check_flag_fail(&flag, "CVDlsSetJacFn", 1);
464 
465  // Set a function to improve guesses for y sent to the linear solver
466  flag = CVodeSetDlsGuessHelper(sd->cvode_mem, guess_helper);
467  check_flag_fail(&flag, "CVodeSetDlsGuessHelper", 1);
468 
469 // Allocate Jacobian on GPU
470 #ifdef CAMP_USE_GPU
471  allocate_jac_gpu(sd->model_data.n_per_cell_solver_jac_elem, n_cells);
472 #endif
473 
474 // Set gpu rxn values
475 #ifdef CAMP_USE_GPU
476  solver_set_rxn_data_gpu(&(sd->model_data));
477 #endif
478 
479 #ifndef FAILURE_DETAIL
480  // Set a custom error handling function
481  flag = CVodeSetErrHandlerFn(sd->cvode_mem, error_handler, (void *)sd);
482  check_flag_fail(&flag, "CVodeSetErrHandlerFn", 0);
483 #endif
484 
485 #endif
486 }

528  * \param solver_data A pointer to the initialized solver data

536 int solver_run(void *solver_data, double *state, double *env, double t_initial,
537  double t_final) {
538 #ifdef CAMP_USE_SUNDIALS
539  SolverData *sd = (SolverData *)solver_data;

828  *
829  * \param t Current model time (s)
830  * \param y Dependent variable array
831  * \param deriv Time derivative vector f(t,y) to calculate
832  * \param solver_data Pointer to the solver data
833  * \return Status code
834  */
835 int f(realtype t, N_Vector y, N_Vector deriv, void *solver_data) {
836  SolverData *sd = (SolverData *)solver_data;
837  ModelData *md = &(sd->model_data);
838  realtype time_step;
839 
840 #ifdef CAMP_DEBUG
841  sd->counterDeriv++;
842 #endif
843 
844  // Get a pointer to the derivative data
845  double *deriv_data = N_VGetArrayPointer(deriv);
846 
847  // Get a pointer to the Jacobian estimated derivative data
848  double *jac_deriv_data = N_VGetArrayPointer(md->J_tmp);
849 
850  // Get the grid cell dimensions
851  int n_cells = md->n_cells;
852  int n_state_var = md->n_per_cell_state_var;
853  int n_dep_var = md->n_per_cell_dep_var;
854 
855  // Get the current integrator time step (s)
856  CVodeGetCurrentStep(sd->cvode_mem, &time_step);
857 
858  // On the first call to f(), the time step hasn't been set yet, so use the
859  // default value
860  time_step = time_step > ZERO ? time_step : sd->init_time_step;
861 
862  // Update the state array with the current dependent variable values.
863  // Signal a recoverable error (positive return value) for negative
864  // concentrations.
865  if (camp_solver_update_model_state(y, md, ZERO, ZERO) != CAMP_SOLVER_SUCCESS)
866  return 1;

868  // Get the Jacobian-estimated derivative
869  N_VLinearSum(1.0, y, -1.0, md->J_state, md->J_tmp);
870  SUNMatMatvec(md->J_solver, md->J_tmp, md->J_tmp2);
871  N_VLinearSum(1.0, md->J_deriv, 1.0, md->J_tmp2, md->J_tmp);
872 
873 #ifdef CAMP_DEBUG
874  // Measure calc_deriv time execution
875  clock_t start = clock();
876 #endif
877 
878 #ifdef CAMP_USE_GPU
879  // Reset the derivative vector
880  N_VConst(ZERO, deriv);
881 
882  // Calculate the time derivative f(t,y)
883  // (this is for all grid cells at once)
884  rxn_calc_deriv_gpu(md, deriv, (double)time_step);
885 #endif
886 
887 #ifdef CAMP_DEBUG
888  clock_t end = clock();
889  sd->timeDeriv += (end - start);
890 #endif
891 
892  // Loop through the grid cells and update the derivative array
893  for (int i_cell = 0; i_cell < n_cells; ++i_cell) {
894  // Set the grid cell state pointers
895  md->grid_cell_id = i_cell;
896  md->grid_cell_state = &(md->total_state[i_cell * n_state_var]);
897  md->grid_cell_env = &(md->total_env[i_cell * CAMP_NUM_ENV_PARAM_]);

901  &(md->aero_rep_env_data[i_cell * md->n_aero_rep_env_data]);
902  md->grid_cell_sub_model_env_data =
903  &(md->sub_model_env_data[i_cell * md->n_sub_model_env_data]);
904 
905  // Update the aerosol representations
906  aero_rep_update_state(md);
907 
908  // Run the sub models
909  sub_model_calculate(md);
910 
911 #ifdef CAMP_DEBUG
912  // Measure calc_deriv time execution
913  clock_t start2 = clock();
914 #endif
915 
916 #ifndef CAMP_USE_GPU
917  // Reset the TimeDerivative
918  time_derivative_reset(sd->time_deriv);
919 
920  // Calculate the time derivative f(t,y)
921  rxn_calc_deriv(md, sd->time_deriv, (double)time_step);
922 
923  // Update the deriv array
924  if (sd->use_deriv_est == 1) {
925  time_derivative_output(sd->time_deriv, deriv_data, jac_deriv_data,
926  sd->output_precision);
927  } else {
928  time_derivative_output(sd->time_deriv, deriv_data, NULL,
929  sd->output_precision);
930  }
931 #else
932  // Add contributions from reactions not implemented on GPU
933  // FIXME need to fix this to use TimeDerivative
934  rxn_calc_deriv_specific_types(md, sd->time_deriv, (double)time_step);
935 #endif
936 
937 #ifdef CAMP_DEBUG
938  clock_t end2 = clock();
939  sd->timeDeriv += (end2 - start2);
940  sd->max_loss_precision = time_derivative_max_loss_precision(sd->time_deriv);
941 #endif
942 
943  // Advance the derivative for the next cell
944  deriv_data += n_dep_var;
945  jac_deriv_data += n_dep_var;
946  }
947 
948  // Return 0 if success
949  return (0);

1282 /** \brief Try to improve guesses of y sent to the linear solver
1283  *
1284  * This function checks if there are any negative guessed concentrations,
1285  * and if there are it calculates a set of initial corrections to the
1286  * guessed state using the state at time \f$t_{n-1}\f$ and the derivative
1287  * \f$f_{n-1}\f$ and advancing the state according to:
1288  * \f[
1289  * y_n = y_{n-1} + \sum_{j=1}^m h_j * f_j
1290  * \f]
1291  * where \f$h_j\f$ is the largest timestep possible where
1292  * \f[
1293  * y_{j-1} + h_j * f_j > 0
1294  * \f]
1295  * and
1296  * \f[
1297  * t_n = t_{n-1} + \sum_{j=1}^m h_j
1298  * \f]
1299  *
1300  * \param t_n Current time [s]
1301  * \param h_n Current time step size [s] If this is set to zero, the change hf
1302  * is assumed to be an adjustment where y_n = y_n1 + hf
1303  * \param y_n Current guess for \f$y(t_n)\f$
1304  * \param y_n1 \f$y(t_{n-1})\f$
1305  * \param hf Current guess for change in \f$y\f$ from \f$t_{n-1}\f$ to
1306  * \f$t_n\f$ [input/output]
1307  * \param solver_data Solver data
1308  * \param tmp1 Temporary vector for calculations
1309  * \param corr Vector of calculated adjustments to \f$y(t_n)\f$ [output]
1310  * \return 1 if corrections were calculated, 0 if not
1311  */
1312 int guess_helper(const realtype t_n, const realtype h_n, N_Vector y_n,
1313  N_Vector y_n1, N_Vector hf, void *solver_data, N_Vector tmp1,
1314  N_Vector corr) {
1315  SolverData *sd = (SolverData *)solver_data;
1316  realtype *ay_n = NV_DATA_S(y_n);
1317  realtype *ay_n1 = NV_DATA_S(y_n1);
1318  realtype *atmp1 = NV_DATA_S(tmp1);
1319  realtype *acorr = NV_DATA_S(corr);
1320  realtype *ahf = NV_DATA_S(hf);
1321  int n_elem = NV_LENGTH_S(y_n);
1322 
1323  // Only try improvements when negative concentrations are predicted
1324  if (N_VMin(y_n) > -SMALL) return 0;
1325 
1326  CAMP_DEBUG_PRINT_FULL("Trying to improve guess");
1327 
1328  // Copy \f$y(t_{n-1})\f$ to working array
1329  N_VScale(ONE, y_n1, tmp1);
1330 
1331  // Get \f$f(t_{n-1})\f$
1332  if (h_n > ZERO) {
1333  N_VScale(ONE / h_n, hf, corr);
1334  } else {
1335  N_VScale(ONE, hf, corr);
1336  }
1337  CAMP_DEBUG_PRINT("Got f0");
1338 
1339  // Advance state interatively
1340  realtype t_0 = h_n > ZERO ? t_n - h_n : t_n - ONE;
1341  realtype t_j = ZERO;
1342  int iter = 0;
1343  for (; iter < GUESS_MAX_ITER && t_0 + t_j < t_n; iter++) {
1344  // Calculate \f$h_j\f$
1345  realtype h_j = t_n - (t_0 + t_j);
1346  int i_fast = -1;
1347  for (int i = 0; i < n_elem; i++) {
1348  realtype t_star = -atmp1[i] / acorr[i];
1349  if ((t_star > ZERO || (t_star == ZERO && acorr[i] < ZERO)) &&
1350  t_star < h_j) {
1351  h_j = t_star;
1352  i_fast = i;
1353  }
1354  }
1355 
1356  // Scale incomplete jumps
1357  if (i_fast >= 0 && h_n > ZERO)
1358  h_j *= 0.95 + 0.1 * rand() / (double)RAND_MAX;
1359  h_j = t_n < t_0 + t_j + h_j ? t_n - (t_0 + t_j) : h_j;
1360 
1361  // Only make small changes to adjustment vectors used in Newton iteration
1362  if (h_n == ZERO &&
1363  t_n - (h_j + t_j + t_0) > ((CVodeMem)sd->cvode_mem)->cv_reltol)
1364  return -1;
1365 
1366  // Advance the state
1367  N_VLinearSum(ONE, tmp1, h_j, corr, tmp1);
1368  CAMP_DEBUG_PRINT_FULL("Advanced state");
1369 
1370  // Advance t_j
1371  t_j += h_j;
1372 
1373  // Recalculate the time derivative \f$f(t_j)\f$
1374  if (f(t_0 + t_j, tmp1, corr, solver_data) != 0) {
1375  CAMP_DEBUG_PRINT("Unexpected failure in guess helper!");
1376  N_VConst(ZERO, corr);
1377  return -1;
1378  }
1379  ((CVodeMem)sd->cvode_mem)->cv_nfe++;
1380 
1381  if (iter == GUESS_MAX_ITER - 1 && t_0 + t_j < t_n) {
1382  CAMP_DEBUG_PRINT("Max guess iterations reached!");
1383  if (h_n == ZERO) return -1;
1384  }
1385  }
1386 
1387  CAMP_DEBUG_PRINT_INT("Guessed y_h in steps:", iter);
1388 
1389  // Set the correction vector
1390  N_VLinearSum(ONE, tmp1, -ONE, y_n, corr);
1391 
1392  // Scale the initial corrections
1393  if (h_n > ZERO) N_VScale(0.999, corr, corr);
1394 
1395  // Update the hf vector
1396  N_VLinearSum(ONE, tmp1, -ONE, y_n1, hf);
1397 
1398  return 1;
1399 }

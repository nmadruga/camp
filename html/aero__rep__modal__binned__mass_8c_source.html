171 /** \brief Update aerosol representation data for a new state
172  *
173  * The modal mass aerosol representation recalculates effective radius and
174  * number concentration for each new state.
175  *
176  * \param model_data Pointer to the model data, including the state array
177  * \param aero_rep_int_data Pointer to the aerosol representation integer data
178  * \param aero_rep_float_data Pointer to the aerosol representation
179  * floating-point data
180  * \param aero_rep_env_data Pointer to the aerosol representation
181  * environment-dependent parameters
182  */
183 void aero_rep_modal_binned_mass_update_state(ModelData *model_data,
184  int *aero_rep_int_data,
185  double *aero_rep_float_data,
186  double *aero_rep_env_data) {
187  int *int_data = aero_rep_int_data;
188  double *float_data = aero_rep_float_data;
189 
190  // Loop through the modes and calculate effective radius and number
191  // concentration
192  for (int i_section = 0; i_section < NUM_SECTION_; i_section++) {
193  double volume, mass;
194  switch (SECTION_TYPE_(i_section)) {
195  // Mode
196  case (MODAL):
197 
198  // Sum the volumes of each species in the mode [m3 m-3]
199  volume = 0.0;
200  for (int i_phase = 0; i_phase < NUM_PHASE_(i_section); i_phase++) {
201  // Get a pointer to the phase on the state array
202  double *state = (double *)(model_data->grid_cell_state);
203  state += PHASE_STATE_ID_(i_section, i_phase, 0);
204 
205  // Set the aerosol-phase mass [kg m-3] and average MW [kg mol-1]
206  aero_phase_get_mass__kg_m3(
207  model_data, PHASE_MODEL_DATA_ID_(i_section, i_phase, 0), state,
208  &(PHASE_MASS_(i_section, i_phase, 0)),
209  &(PHASE_AVG_MW_(i_section, i_phase, 0)), NULL, NULL);
210 
211  // Get the phase volume [m3 m-3]
212  double phase_volume = 0.0;
213  aero_phase_get_volume__m3_m3(
214  model_data, PHASE_MODEL_DATA_ID_(i_section, i_phase, 0), state,
215  &phase_volume, NULL);
216  volume += phase_volume;
217  }
218 
219  // Calculate the number concentration [# m-3] based on the total mode
220  // volume (see aero_rep_modal_binned_mass_get_number_conc for details)
221  NUMBER_CONC_(i_section, 0) =
222  volume * 6.0 /
223  (M_PI * pow(GMD_(i_section), 3) *
224  exp(9.0 / 2.0 * pow(log(GSD_(i_section)), 2)));
225 
226  break;
227 
228  // Bins
229  case (BINNED):
230 
231  // Loop through the bins
232  for (int i_bin = 0; i_bin < NUM_BINS_(i_section); i_bin++) {
233  // Sum the volumes of each species in the bin [m3 m-3]
234  volume = 0.0;
235  for (int i_phase = 0; i_phase < NUM_PHASE_(i_section); i_phase++) {
236  // Get a pointer to the phase on the state array
237  double *state = (double *)(model_data->grid_cell_state);
238  state += PHASE_STATE_ID_(i_section, i_phase, i_bin);
239 
240  // Set the aerosol-phase mass [kg m-3] and average MW [kg mol-1]
241  aero_phase_get_mass__kg_m3(
242  model_data, PHASE_MODEL_DATA_ID_(i_section, i_phase, i_bin),
243  state, &(PHASE_MASS_(i_section, i_phase, i_bin)),
244  &(PHASE_AVG_MW_(i_section, i_phase, i_bin)), NULL, NULL);
245 
246  // Get the phase volume [m3 m-3]
247  double phase_volume = 0.0;
248  aero_phase_get_volume__m3_m3(
249  model_data, PHASE_MODEL_DATA_ID_(i_section, i_phase, i_bin),
250  state, &phase_volume, NULL);
251  volume += phase_volume;
252  }
253 
254  // Calculate the number concentration [# m-3] based on the total bin
255  // volume (see aero_rep_modal_binned_mass_get_number_conc for details)
256  NUMBER_CONC_(i_section, i_bin) =
257  volume * 3.0 / (4.0 * M_PI) /
258  pow(BIN_DP_(i_section, i_bin) / 2.0, 3);
259  }
260 
261  break;
262  }
263  }
264 
265  return;
266 }

268 /** \brief Get the effective particle radius \f$r_{eff}\f$ (m)
269  *
270  * The modal mass effective radius is calculated for a log-normal distribution
271  * where the geometric mean diameter (\f$\tilde{D}_n\f$) and geometric standard
272  * deviation (\f$\tilde{\sigma}_g\f$) are set by the aerosol model prior to
273  * solving the chemistry. Thus, all \f$\frac{\partial r_{eff}}{\partial y}\f$
274  * are zero. The effective radius is calculated according to the equation given
275  * in Table 1 of Zender \cite Zender2002 :
276  *
277  * \f[
278  * r_{eff} = \frac{\tilde{D}_n}{2}*exp(9 ln(\tilde{\sigma}_g)^2/2)
279  * \f]
280  * \f[
281  * r_{eff} = \frac{D_{eff}}{2}
282  * \f]
283  *
284  * For bins, \f$r_{eff}\f$ is assumed to be the bin radius.
285  *
286  * \param model_data Pointer to the model data, including the state array
287  * \param aero_phase_idx Index of the aerosol phase within the representation
288  * \param radius Effective particle radius (m)
289  * \param partial_deriv \f$\frac{\partial r_{eff}}{\partial y}\f$ where \f$y\f$
290  * are species on the state array
291  * \param aero_rep_int_data Pointer to the aerosol representation integer data
292  * \param aero_rep_float_data Pointer to the aerosol representation
293  * floating-point data
294  * \param aero_rep_env_data Pointer to the aerosol representation
295  * environment-dependent parameters
296  */
297 void aero_rep_modal_binned_mass_get_effective_radius__m(
298  ModelData *model_data, int aero_phase_idx, double *radius,
299  double *partial_deriv, int *aero_rep_int_data, double *aero_rep_float_data,
300  double *aero_rep_env_data) {
301  int *int_data = aero_rep_int_data;
302  double *float_data = aero_rep_float_data;
303 
304  for (int i_section = 0; i_section < NUM_SECTION_; i_section++) {
305  for (int i_bin = 0; i_bin < NUM_BINS_(i_section); i_bin++) {
306  aero_phase_idx -= NUM_PHASE_(i_section);
307  if (aero_phase_idx < 0) {
308  *radius = EFFECTIVE_RADIUS_(i_section, i_bin);
309  // Effective radii are constant for bins and modes
310  if (partial_deriv) {
311  for (int i_phase = 0; i_phase < NUM_PHASE_(i_section); ++i_phase) {
312  for (int i_elem = 0;
313  i_elem < PHASE_NUM_JAC_ELEM_(i_section, i_phase, i_bin);
314  ++i_elem) {
315  *(partial_deriv++) = ZERO;
316  }
317  }
318  }
319  i_section = NUM_SECTION_;
320  break;
321  }
322  }
323  }
324 
325  return;
326 }

328 /** \brief Get the particle number concentration \f$n\f$
329  * (\f$\mbox{\si{\#\per\cubic\metre}}\f$)
330  *
331  * The modal mass number concentration is calculated for a log-normal
332  * distribution where the geometric mean diameter (\f$\tilde{D}_n\f$) and
333  * geometric standard deviation (\f$\tilde{\sigma}_g\f$) are set by the aerosol
334  * model prior to solving the chemistry. The number concentration is
335  * calculated according to the equation given in Table 1 of Zender
336  * \cite Zender2002 :
337  * \f[
338  * n = N_0 = \frac{6V_0}{\pi}\tilde{D}_n^{-3}e^{-9
339  * ln(\tilde{\sigma}_g)^2/2} \f] \f[ V_0 = \sum_i{\frac{m_i}{\rho_i}} \f] where
340  * \f$\rho_i\f$ and \f$m_i\f$ are the density and total mass of species \f$i\f$
341  * in the specified mode.
342  *
343  * The binned number concentration is calculated according to:
344  * \f[
345  * n = V_0 / V_p
346  * \f]
347  * \f[
348  * V_p = \frac{4}{3}\pi r^{3}
349  * \f]
350  * where \f$r\f$ is the radius of the size bin and \f$V_0\f$ is defined as
351  * above.
352  *
353  * \param model_data Pointer to the model data, including the state array
354  * \param aero_phase_idx Index of the aerosol phase within the representation
355  * \param number_conc Particle number concentration, \f$n\f$
356  * (\f$\mbox{\si{\#\per\cubic\centi\metre}}\f$)
357  * \param partial_deriv \f$\frac{\partial n}{\partial y}\f$ where \f$y\f$ are
358  * the species on the state array
359  * \param aero_rep_int_data Pointer to the aerosol representation integer data
360  * \param aero_rep_float_data Pointer to the aerosol representation
361  * floating-point data
362  * \param aero_rep_env_data Pointer to the aerosol representation
363  * environment-dependent parameters
364  */
365 void aero_rep_modal_binned_mass_get_number_conc__n_m3(
366  ModelData *model_data, int aero_phase_idx, double *number_conc,
367  double *partial_deriv, int *aero_rep_int_data, double *aero_rep_float_data,
368  double *aero_rep_env_data) {
369  int *int_data = aero_rep_int_data;
370  double *float_data = aero_rep_float_data;
371 
372  for (int i_section = 0; i_section < NUM_SECTION_ && aero_phase_idx >= 0;
373  i_section++) {
374  for (int i_bin = 0; i_bin < NUM_BINS_(i_section) && aero_phase_idx >= 0;
375  i_bin++) {
376  aero_phase_idx -= NUM_PHASE_(i_section);
377  if (aero_phase_idx < 0) {
378  *number_conc = NUMBER_CONC_(i_section, i_bin);
379  if (partial_deriv) {
380  for (int i_phase = 0; i_phase < NUM_PHASE_(i_section); ++i_phase) {
381  // Get a pointer to the phase on the state array
382  double *state = (double *)(model_data->grid_cell_state);
383  state += PHASE_STATE_ID_(i_section, i_phase, i_bin);
384 
385  // Get the aerosol phase volume [m3 m-3]
386  double phase_volume = 0.0;
387  aero_phase_get_volume__m3_m3(
388  model_data, PHASE_MODEL_DATA_ID_(i_section, i_phase, i_bin),
389  state, &phase_volume, partial_deriv);
390 
391  // Convert d_vol/d_conc to d_number/d_conc
392  for (int i_elem = 0;
393  i_elem < PHASE_NUM_JAC_ELEM_(i_section, i_phase, i_bin);
394  ++i_elem) {
395  switch (SECTION_TYPE_(i_section)) {
396  case (MODAL):
397  *(partial_deriv++) *=
398  6.0 / (M_PI * pow(GMD_(i_section), 3) *
399  exp(9.0 / 2.0 * pow(log(GSD_(i_section)), 2)));
400  break;
401  case (BINNED):
402  *(partial_deriv++) *= 3.0 / (4.0 * M_PI) /
403  pow(BIN_DP_(i_section, i_bin) / 2.0, 3);
404  break;
405  }
406  }
407  }
408  }
409  i_section = NUM_SECTION_;
410  break;
411  }
412  }
413  }
414 
415  return;
416 }

528 
529  for (int i_section = 0; i_section < NUM_SECTION_ && aero_phase_idx >= 0;
530  ++i_section) {
531  for (int i_bin = 0; i_bin < NUM_BINS_(i_section) && aero_phase_idx >= 0;
532  ++i_bin) {
533  if (aero_phase_idx < 0 || aero_phase_idx >= NUM_PHASE_(i_section)) {
534  aero_phase_idx -= NUM_PHASE_(i_section);
535  continue;
536  }
537  for (int i_phase = 0; i_phase < NUM_PHASE_(i_section); ++i_phase) {
538  if (aero_phase_idx == 0) {
539  *aero_phase_avg_MW = PHASE_AVG_MW_(i_section, i_phase, i_bin);
540  if (partial_deriv) {
541  // Get a pointer to the phase on the state array
542  double *state = (double *)(model_data->grid_cell_state);
543  state += PHASE_STATE_ID_(i_section, i_phase, i_bin);
544 
545  // Get d_MW / d_conc
546  double mass, mw;
547  aero_phase_get_mass__kg_m3(
548  model_data, PHASE_MODEL_DATA_ID_(i_section, i_phase, i_bin),
549  state, &mass, &mw, NULL, partial_deriv);
550  partial_deriv += PHASE_NUM_JAC_ELEM_(i_section, i_phase, i_bin);
551  }
552 
553  // Other phases present in the bin/mode do not contribute to the
554  // average MW of the aerosol phase
555  } else if (partial_deriv) {
556  for (int i_elem = 0;
557  i_elem < PHASE_NUM_JAC_ELEM_(i_section, i_phase, i_bin);
558  ++i_elem) {
559  *(partial_deriv++) = ZERO;
560  }
561  }
562  aero_phase_idx -= 1;
563  }
564  }
565  }

570 /** \brief Update the aerosol representation data
571  *
572  * The model mass aerosol representation update data is structured as follows:
573  *
574  * - \b int aero_rep_id (Id of one or more aerosol representations set by the
575  * host model using the
576  * camp_aero_rep_modal_binned_mass::aero_rep_modal_binned_mass_t::set_id
577  * function prior to initializing the solver.)
578  * - \b int update_type (Type of update to perform. Can be UPDATE_GMD or
579  * UPDATE_GSD.)
580  * - \b int section_id (Index of the mode to update.)
581  * - \b double new_value (Either the new GMD (m) or the new GSD (unitless).)
582  *
583  * \param update_data Pointer to the updated aerosol representation data
584  * \param aero_rep_int_data Pointer to the aerosol representation integer data
585  * \param aero_rep_float_data Pointer to the aerosol representation
586  * floating-point data
587  * \param aero_rep_env_data Pointer to the aerosol representation
588  * environment-dependent parameters
589  * \return Flag indicating whether this is the aerosol representation to update
590  */
591 bool aero_rep_modal_binned_mass_update_data(void *update_data,
592  int *aero_rep_int_data,
593  double *aero_rep_float_data,
594  double *aero_rep_env_data) {
595  int *int_data = aero_rep_int_data;
596  double *float_data = aero_rep_float_data;
597 
598  int *aero_rep_id = (int *)update_data;
599  int *update_type = (int *)&(aero_rep_id[1]);
600  int *section_id = (int *)&(update_type[1]);
601  double *new_value = (double *)&(section_id[1]);
602  bool ret_val = false;
603 
604  // Set the new GMD or GSD for matching aerosol representations
605  if (*aero_rep_id == AERO_REP_ID_ && AERO_REP_ID_ != 0) {
606  if (*update_type == UPDATE_GMD) {
607  if (SECTION_TYPE_(*section_id) != MODAL) {
608  printf(
609  "\n\nERROR Trying to set geometric mean diameter for non-modal"
610  " aerosol section.");
611  exit(1);
612  }
613  GMD_(*section_id) = (double)*new_value; // [m]
614  ret_val = true;
615  } else if (*update_type == UPDATE_GSD) {
616  if (SECTION_TYPE_(*section_id) != MODAL) {
617  printf(
618  "\n\nERROR Trying to set geometric standard deviation for non-modal"
619  " aerosol section.");
620  exit(1);
621  }
622  GSD_(*section_id) = (double)*new_value;
623  ret_val = true;
624  }
625  }
626 
627  if (ret_val == true) {
628  /// Recalculate the effective radius [m]
629  ///
630  /// Equation based on \cite Zender2002 eq. (23) and Table 1 median diameter
631  /// (\f$\tilde{D_n}\f$)
632  /// and Table 1 effective radius \f$(D_s, D_eff\f$) equations:
633  /// \f[
634  /// \tilde{\sigma_g} \equiv ln( \sigma_g )
635  /// \f]
636  /// \f[
637  /// \tilde{D_n} = \bar{D_n} e^{-\tilde{\sigma_g}^2 / 2}
638  /// \f]
639  /// \f[
640  /// D_s = D_{eff} = \tilde{D_n} e^{5 \tilde{\sigma_g}^2 / 2}
641  /// \f]
642  /// \f[
643  /// D_{eff} = \bar{D_n} e^{ 2 \tilde{\sigma_g}^2 }
644  /// \f]
645  /// where \f$\bar{D_n}\f$ is the mean diameter [m] and \f$\sigma_g\f$
646  /// is the geometric standard deviation [unitless].
647  ///
648  double ln_gsd = log(GSD_(*section_id));
649  EFFECTIVE_RADIUS_(*section_id, 0) =
650  GMD_(*section_id) / 2.0 * exp(2.0 * ln_gsd * ln_gsd);
651  }
652 
653  return ret_val;
654 }

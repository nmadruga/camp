8!> \page camp_aero_rep_single_particle CAMP: Single Particle Aerosol Representation
9!!
10!! The single particle aerosol representation is for use with a PartMC
11!! particle-resolved run. The \c json object for this \ref camp_aero_rep
12!! "aerosol representation" has the following format:
13!! \code{.json}
14!! { "camp-data" : [
15!! {
16!! "name" : "my single particle aero rep",
17!! "type" : "AERO_REP_SINGLE_PARTICLE"
18!! },
19!! ...
20!! ]}
21!! \endcode
22!! The key-value pair \b type is required and must be \b
23!! AERO_REP_SINGLE_PARTICLE. This representation assumes that every \ref
24!! input_format_aero_phase "aerosol phase" available will be present
25!! once in each particle.
26!!
27!! The number concentration for each particle must be
28!! set from an external model using
29!! \c camp_aero_rep_single_particle::aero_rep_update_data_single_particle_number_t
30!! objects.
31

216 !> Initialize the aerosol representation data, validating component data and
217 !! loading any required information from the \c
218 !! aero_rep_data_t::property_set. This routine should be called once for
219 !! each aerosol representation at the beginning of a model run after all
220 !! the input files have been read in. It ensures all data required during
221 !! the model run are included in the condensed data arrays.
222 subroutine initialize(this, aero_phase_set, spec_state_id)
223
224 !> Aerosol representation data
225 class(aero_rep_single_particle_t), intent(inout) :: this
226 !> The set of aerosol phases
227 type(aero_phase_data_ptr), pointer, intent(in) :: aero_phase_set(:)
228 !> Beginning state id for this aerosol representationin the model species
229 !! state array
230 integer(kind=i_kind), intent(in) :: spec_state_id
231
232 character(len=:), allocatable :: key_name
233 integer(kind=i_kind) :: i_particle, i_phase, curr_id
234 integer(kind=i_kind) :: num_int_param, num_float_param, num_particles
235
236 ! Start off the counters
237 num_int_param = num_int_prop_ + 3*size(aero_phase_set)
238 num_float_param = num_real_prop_
239
240 ! Get the maximum number of computational particles
241 key_name = "maximum computational particles"
242 call assert_msg(857908074, &
243 this%property_set%get_int(key_name, num_particles), &
244 "Missing maximum number of computational particles")
245
246 ! Assume all phases will be applied once to each particle
247 allocate(this%aero_phase(size(aero_phase_set)*num_particles))
248 do i_particle = 1, num_particles
249 do i_phase = 1, size(aero_phase_set)
250 this%aero_phase((i_particle-1)*size(aero_phase_set)+i_phase) = &
251 aero_phase_set(i_phase)
252 end do
253 end do
254
255 ! Allocate condensed data arrays
256 allocate(this%condensed_data_int(num_int_param))
257 allocate(this%condensed_data_real(num_float_param))
258 this%condensed_data_int(:) = int(0, kind=i_kind)
259 this%condensed_data_real(:) = real(0.0, kind=dp)
260
261 ! Save the number of computational particles
262 max_particles_ = num_particles
263
264 ! Save space for the environment-dependent parameters
265 this%num_env_params = num_env_param_per_particle_ * num_particles
266
267 ! Set phase state and model data ids
268 num_phase_ = size(aero_phase_set)
269 this%state_id_start = spec_state_id
270 curr_id = spec_state_id
271 do i_phase = 1, num_phase_
272 phase_state_id_(i_phase) = curr_id
273 phase_model_data_id_(i_phase) = i_phase
274 curr_id = curr_id + aero_phase_set(i_phase)%val%size()
275 end do
276 particle_state_size_ = curr_id - spec_state_id
277
278 ! Initialize the aerosol representation id
279 aero_rep_id_ = -1
280
281 ! Set the unique names for the chemical species
282 this%unique_names_ = this%unique_names( )
283
284 end subroutine initialize

336!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
337
338 !> Get a list of unique names for each element on the
339 !! \c camp_camp_state::camp_state_t::state_var array for this aerosol
340 !! representation. The list may be restricted to a particular phase and/or
341 !! aerosol species by including the phase_name and spec_name arguments.
342 !!
343 !! For a single particle representation, the unique names will be a 'P'
344 !! followed by the computational particle number, a '.', the phase name,
345 !! another '.', and the species name.
346 function unique_names(this, phase_name, tracer_type, spec_name)
347
348 use camp_util, only : integer_to_string
349
350 !> List of unique names
351 type(string_t), allocatable :: unique_names(:)
352 !> Aerosol representation data
353 class(aero_rep_single_particle_t), intent(in) :: this
354 !> Aerosol phase name
355 character(len=*), optional, intent(in) :: phase_name
356 !> Aerosol-phase species tracer type
357 integer(kind=i_kind), optional, intent(in) :: tracer_type
358 !> Aerosol-phase species name
359 character(len=*), optional, intent(in) :: spec_name
360
361 integer(kind=i_kind) :: num_spec, i_part, i_spec, j_spec, i_phase
362 integer(kind=i_kind) :: curr_tracer_type
363 character(len=:), allocatable :: curr_phase_name
364 type(string_t), allocatable :: spec_names(:)
365
366 ! Copy saved unique names when no filters are included
367 if (.not. present(phase_name) .and. &
368 .not. present(tracer_type) .and. &
369 .not. present(spec_name) .and. &
370 allocated(this%unique_names_)) then
371 unique_names = this%unique_names_
372 return
373 end if
374
375 ! Count the number of unique names
376 num_spec = 0
377 do i_phase = 1, num_phase_
378 curr_phase_name = this%aero_phase(i_phase)%val%name()
379 if (present(phase_name)) then
380 if (phase_name.ne.curr_phase_name) cycle
381 end if
382 if (present(spec_name).or.present(tracer_type)) then
383 spec_names = this%aero_phase(i_phase)%val%get_species_names()
384 do j_spec = 1, size(spec_names)
385 curr_tracer_type = &
386 this%aero_phase(i_phase)%val%get_species_type( &
387 spec_names(j_spec)%string)
388 if (present(spec_name)) then
389 if (spec_name.ne.spec_names(j_spec)%string) cycle
390 end if
391 if (present(tracer_type)) then
392 if (tracer_type.ne.curr_tracer_type) cycle
393 end if
394 num_spec = num_spec + 1
395 end do
396 deallocate(spec_names)
397 else
398 num_spec = num_spec + this%aero_phase(i_phase)%val%size()
399 end if
400 end do
401
402 ! Allocate space for the unique names and assign them
403 allocate(unique_names(num_spec*max_particles_))
404 i_spec = 1
405 do i_part = 1, max_particles_
406 do i_phase = 1, num_phase_
407 curr_phase_name = this%aero_phase(i_phase)%val%name()
408 if (present(phase_name)) then
409 if (phase_name.ne.curr_phase_name) cycle
410 end if
411 spec_names = this%aero_phase(i_phase)%val%get_species_names()
412 num_spec = this%aero_phase(i_phase)%val%size()
413 do j_spec = 1, num_spec
414 curr_tracer_type = &
415 this%aero_phase(i_phase)%val%get_species_type( &
416 spec_names(j_spec)%string)
417 if (present(spec_name)) then
418 if (spec_name.ne.spec_names(j_spec)%string) cycle
419 end if
420 if (present(tracer_type)) then
421 if (tracer_type.ne.curr_tracer_type) cycle
422 end if
423 unique_names(i_spec)%string = 'P'//trim(integer_to_string(i_part))//&
424 "."//curr_phase_name//'.'//spec_names(j_spec)%string
425 i_spec = i_spec + 1
426 end do
427 end do
428 deallocate(spec_names)
429 end do
430
431 end function unique_names

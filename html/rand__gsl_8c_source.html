38 /** \brief Initialize the random number generator with the given seed.
39  *
40  * This must be called before any other GSL random number functions
41  * are called.
42  *
43  * \param seed The random seed to use.
44  * \return CAMP_RAND_GSL_SUCCESS on success, otherwise an error code.
45  * \sa camp_rand_finalize_gsl() to cleanup the generator.
46  */
47 int camp_srand_gsl(int seed)
48 {
49  if (camp_rand_gsl_rng) {
50  return CAMP_RAND_GSL_ALREADY_INIT;
51  }
52  gsl_set_error_handler_off(); // turn off automatic error handling
53  camp_rand_gsl_rng = gsl_rng_alloc(gsl_rng_mt19937);
54  if (camp_rand_gsl_rng == NULL) {
55  return CAMP_RAND_GSL_INIT_FAIL;
56  }
57  gsl_rng_set(camp_rand_gsl_rng, seed);
58  return CAMP_RAND_GSL_SUCCESS;
59 }
60 
61 /** \brief Cleanup and deallocate the random number generator.
62  *
63  * This must be called after camp_srand_gsl().
64  *
65  * \return CAMP_RAND_GSL_SUCCESS on success, otherwise an error code.
66  */
67 int camp_rand_finalize_gsl()
68 {
69  if (!camp_rand_gsl_rng) {
70  return CAMP_RAND_GSL_NOT_INIT;
71  }
72  gsl_rng_free(camp_rand_gsl_rng);
73  camp_rand_gsl_rng = NULL;
74  return CAMP_RAND_GSL_SUCCESS;
75 }

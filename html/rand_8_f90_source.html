234  !> Generate a Poisson-distributed random number with the given
235  !> mean.
236  !!
237  !! See http://en.wikipedia.org/wiki/Poisson_distribution for
238  !! information on the method. The method used at present is rather
239  !! inefficient and inaccurate (brute force for mean below 10 and
240  !! normal approximation above that point).
241  !!
242  !! The best known method appears to be due to Ahrens and Dieter (ACM
243  !! Trans. Math. Software, 8(2), 163-179, 1982) and is available (in
244  !! various forms) from:
245  !! - http://www.netlib.org/toms/599
246  !! - http://www.netlib.org/random/ranlib.f.tar.gz
247  !! - http://users.bigpond.net.au/amiller/random/random.f90
248  !! - http://www.netlib.org/random/random.f90
249  !!
250  !! Unfortunately the above code is under the non-free license:
251  !! - http://www.acm.org/pubs/copyright_policy/softwareCRnotice.html

254  !! Variate Generation", Springer-Verlag, 1986.

312  !> Generate a Binomial-distributed random number with the given
313  !> parameters.
314  !!
315  !! See http://en.wikipedia.org/wiki/Binomial_distribution for
316  !! information on the method. The method used at present is rather
317  !! inefficient and inaccurate (brute force for \f$np \le 10\f$ and
318  !! \f$n(1-p) \le 10\f$, otherwise normal approximation).
319  !!
320  !! For better methods, see L. Devroye, "Non-Uniform Random Variate
321  !! Generation", Springer-Verlag, 1986.
322  integer function rand_binomial(n, p)

325  integer, intent(in) :: n
326  !> Sample probability.
327  real(kind=dp), intent(in) :: p
328 
329 #ifdef CAMP_USE_GSL
330  integer(kind=c_int) :: n_c
331  real(kind=c_double) :: p_c
332  integer(kind=c_int), target :: harvest
333  type(c_ptr) :: harvest_ptr
334 #else
335  real(kind=dp) :: np, nomp, q, g_real
336  integer :: k, g, sum
337 #endif
338 
339 #ifdef CAMP_USE_GSL
340 #ifndef DOXYGEN_SKIP_DOC
341  interface
342  integer(kind=c_int) function camp_rand_binomial_gsl(n, p, harvest) &
343  bind(c)
344  use iso_c_binding
345  integer(kind=c_int), value :: n
346  real(kind=c_double), value :: p
347  type(c_ptr), value :: harvest
348  end function camp_rand_binomial_gsl
349  end interface
350 #endif
351 #endif
352 
353  call assert(130699849, n >= 0)
354  call assert(754379796, p >= 0d0)
355  call assert(678506029, p <= 1d0)
356 #ifdef CAMP_USE_GSL
357  n_c = int(n, kind=c_int)
358  p_c = real(p, kind=c_double)
359  harvest_ptr = c_loc(harvest)
360  call rand_check_gsl(208869397, &
361  camp_rand_binomial_gsl(n_c, p_c, harvest_ptr))
362  rand_binomial = int(harvest)
363 #else
364  np = real(n, kind=dp) * p
365  nomp = real(n, kind=dp) * (1d0 - p)
366  if ((np >= 10d0) .and. (nomp >= 10d0)) then
367  ! normal approximation with continuity correction
368  k = nint(rand_normal(np, sqrt(np * (1d0 - p))))
369  rand_binomial = min(max(k, 0), n)
370  elseif (np < 1d-200) then
371  rand_binomial = 0
372  elseif (nomp < 1d-200) then
373  rand_binomial = n
374  else
375  ! First waiting time method of Devroye (p. 525).
376  ! We want p small, so if p > 1/2 then we compute with 1 - p and
377  ! take n - k at the end.
378  if (p <= 0.5d0) then
379  q = p
380  else
381  q = 1d0 - p
382  end if
383  k = 0
384  sum = 0
385  do
386  ! G is geometric(q)
387  g_real = log(camp_random()) / log(1d0 - q)
388  ! early bailout for cases to avoid integer overflow
389  if (g_real > real(n - sum, kind=dp)) exit
390  g = ceiling(g_real)
391  sum = sum + g
392  if (sum > n) exit
393  k = k + 1
394  end do
395  if (p <= 0.5d0) then
396  rand_binomial = k
397  else
398  rand_binomial = n - k
399  end if
400  call assert(359087410, rand_binomial <= n)
401  end if
402 #endif
403 
404  end function rand_binomial

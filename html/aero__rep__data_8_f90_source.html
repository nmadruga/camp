8 !> \page camp_aero_rep CAMP: Aerosol Representation (general)
9 !!
10 !! An aerosol representation acts as an interface between an aerosol
11 !! micro-physics model and \ref index "CAMP".
12 !! Types that extend the abstract \c aero_rep_data_t type should be developed
13 !! for each type of aerosol representation used by an external model (e.g.,
14 !! binned, modal, single particle).
15 !!
16 !! The available aerosol representations are:
17 !! - \subpage camp_aero_rep_single_particle "Single Particle"
18 !! - \subpage camp_aero_rep_modal_binned_mass "Mass-only Binned/Modal"
19 !!
20 !! The general input format for an aerosol representation can be found
21 !! \subpage input_format_aero_rep "here".
22

379  !> \page input_format_aero_rep Input JSON Object Format: Aerosol Representation (general)
380  !!
381  !! A \c json object containing information about an \ref camp_aero_rep
382  !! "aerosol representation" has the following format:
383  !! \code{.json}
384  !! { "camp-data" : [
385  !! {
386  !! "name" : "my aero rep",
387  !! "type" : "AERO_REP_TYPE",
388  !! "some parameter" : 123.34,
389  !! "some other parameter" : true,
390  !! "nested parameters" : {
391  !! "sub param 1" : 12.43,
392  !! "sub param other" : "some text",
393  !! ...
394  !! },
395  !! ...
396  !! },
397  !! ...
398  !! ]}
399  !! \endcode
400  !! Aerosol representations must have a unique \b name that will be used to
401  !! identify the aerosol representation during initialization. The key-value
402  !! pair \b type is also required and must correspond to a valid aerosol
403  !! representation type. These include:
404  !!
405  !! - \subpage camp_aero_rep_single_particle "AERO_REP_SINGLE_PARTICLE"
406  !! - \subpage camp_aero_rep_modal_binned_mass
407  !! "AERO_REP_MODAL_BINNED_MASS"
408  !!
409  !! All remaining data are optional and may include any valid \c json value.
410  !! However, extending types will have specific requirements for the
411  !! remaining data.
412

415  !> Load an aerosol representation from an input file
416 #ifdef CAMP_USE_JSON
417  subroutine load(this, json, j_obj)
418 
419  !> Aerosol representation data
420  class(aero_rep_data_t), intent(inout) :: this
421  !> JSON core
422  type(json_core), pointer, intent(in) :: json
423  !> JSON object
424  type(json_value), pointer, intent(in) :: j_obj
425 
426  type(json_value), pointer :: child, next
427  character(kind=json_ck, len=:), allocatable :: key, unicode_str_val
428  integer(kind=json_ik) :: var_type
429  logical :: found_name
430 
431  ! allocate space for the aerosol representation property set
432  this%property_set => property_t()
433 
434  if (.not.allocated(this%rep_name)) &
435  this%rep_name = "unknown aerosol representation"
436  found_name = .false.
437 
438  ! cycle through the aerosol representation properties to find the name and
439  ! load the remaining data into the aerosol representation property set
440  next => null()
441  call json%get_child(j_obj, child)
442  do while (associated(child))
443  call json%info(child, name=key, var_type=var_type)
444 
445  ! aerosol representation name
446  if (key.eq."name") then
447  call assert_msg(196193896, var_type.eq.json_string, &
448  "Received non-string value for aerosol rep name")
449  call json%get(child, unicode_str_val)
450  this%rep_name = unicode_str_val
451  found_name = .true.
452 
453  ! load remaining data tinto the property set
454  else if (key.ne."type") then
455  call this%property_set%load(json, child, .false., this%rep_name)
456  end if
457 
458  call json%get_next(child, next)
459  child => next
460  end do
461  call assert_msg(420903951, found_name, &
462  "Received unnamed aerosol representation.")
463 #else
464  subroutine load(this)
465 
466  !> Aerosol representation data
467  class(aero_rep_data_t), intent(inout) :: this
468 
469  call warn_msg(433045149, "No support for input files")
470 #endif
471 
472  end subroutine load
